  <script>
    // Schimbă textul și culoarea la fiecare apăsare
    function schimbaTextul() {
      const paragraf = document.getElementById("text");
      const culori = ["red", "blue", "green", "purple", "orange"];
      const mesaje = [
        "Salut Antonio!",
        "Sper să ai o zi grozavă!",
        "JavaScript e distractiv!",
        "Apasă din nou și vezi ce se întâmplă!",
        "Bravo, funcționează!"
      ];

      // Alege aleatoriu o culoare și un mesaj
      const culoare = culori[Math.floor(Math.random() * culori.length)];
      const mesaj = mesaje[Math.floor(Math.random() * mesaje.length)];

      paragraf.style.color = culoare;
      paragraf.innerHTML = mesaj;
    }
  </script>
